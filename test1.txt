#include <stdio.h> //подкл библиотеки ввода и вывода
#include <windows.h>//подкл библиотеки для кириллицы 
void main()//заголовок функции
{//начало функции
	/*Предусловие: радиус круга вводится с консоли
	Постусловие: программа вычисляет площадь круга по заданному радиусу*/
	SetConsoleCP(1251);//Обращаемся к библиотеки для кириллицы
	SetConsoleOutputCP(1251);
	const double pi = 3.14;//присвоили значение константе пи
	double R, S;//объявление переменных
	printf("Введите значение радиуса ");//текст для пользователя
	scanf("%lf", &R);//вызывается консоль для ввода R
	S = pi * R * R;//формула площади круга
	printf("Площадь круга = %lf", S);//текст для пользователя
}//конец функции
