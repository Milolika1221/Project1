#include <iostream>//потоковая библиотека ввода и вывода
using namespace :: std;  
#include <windows.h> //подключ библиотеки для кириллицы
/*Найти количество чисел последовательности меньших -23 и их сумму Указать номер пятого числа последовательности, меньшего -23.*/
void main()//заголовок функции
{//начало функции
	SetConsoleCP(1251);//обращение к библиотеки для кириллицы
	SetConsoleOutputCP(1251);//обращение к библиотеки для кириллицы
	int n, k = 0, s = 0, N, i = 0;//объявляем целочисленные переменные
	cout << "Введите числовую последовательность через пробел. В конце нажмите Enter.\n"; //оператор вывода текста на экран
	do//начало цикла
	{
		cin >> n;//оператор ввода с клавиатуры значения n
		i += 1; //i - считает к-во чисел
		if (n < -23)//условие: n < -23
		{
			k += 1;// k = k + 1. Считаем к-во нужных нам чисел
			s += n;//считаем сумму нужных нам чисел
			if (k == 5)//если к-во нужных чисел будет равно 5, то будем в перемнной N хранить номер 5 числа
				N = i;//Записываем порядковый номер числа в переменную N
		}//конец условия

	} while (getchar() != '\n');//конец цикла. В конце while проверяет, если была нажата кнопка Enter, то выходим из цикла, иначе продолжается работа
	if (k < 5) //условие. Проверяем к-во чисел. Если их меньше 5, то выведем соответствующее сообщение "К-во чисел меньше пяти"
		cout << "К-во чисел меньше пяти, "; //оператор вывода текста на экран
	else
		cout << "N = " << N << ","; //оператор вывода текста на экран
	cout << " k = " << k << ", " << "s = " << s;//оператор вывода текста на экран
}//конец функции
